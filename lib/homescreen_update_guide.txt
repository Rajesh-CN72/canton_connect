HOW TO UPDATE YOUR HOMESCREEN:

Replace your current HomeScreen build method with this structure:

@override
Widget build(BuildContext context) {
  return Scaffold(
    appBar: const CustomAppBar(
      title: 'Home',
      showBackButton: false,
    ),
    body: SingleChildScrollView(
      child: Column(
        children: [
          // Smart Carousel
          const SizedBox(height: 20),
          SmartCarousel(
            onSectionTap: (sectionId) {
              // Handle section tap
              print('Carousel section tapped: \');
            },
          ),
          
          // Categories Section
          const SizedBox(height: 32),
          Padding(
            padding: const EdgeInsets.symmetric(horizontal: 16),
            child: Text(
              'Categories',
              style: Theme.of(context).textTheme.headlineSmall?.copyWith(
                fontWeight: FontWeight.bold,
              ),
            ),
          ),
          const SizedBox(height: 16),
          SizedBox(
            height: 500, // Adjusted height for 6 categories
            child: CategoryGrid(
              onCategoryTap: (categoryId) {
                // Handle category tap
                print('Category tapped: \');
              },
            ),
          ),
          
          // Featured Chefs Section
          const SizedBox(height: 32),
          Padding(
            padding: const EdgeInsets.symmetric(horizontal: 16),
            child: Text(
              'Featured Chefs',
              style: Theme.of(context).textTheme.headlineSmall?.copyWith(
                fontWeight: FontWeight.bold,
              ),
            ),
          ),
          const SizedBox(height: 16),
          SizedBox(
            height: 400,
            child: FeaturedChefsGrid(
              onChefTap: (chefId) {
                // Handle chef tap
                print('Chef tapped: \');
              },
            ),
          ),
          const SizedBox(height: 20),
        ],
      ),
    ),
    bottomNavigationBar: SmartBottomNavigation(
      currentIndex: _currentBottomNavIndex,
      onTap: (index) {
        setState(() {
          _currentBottomNavIndex = index;
        });
      },
    ),
  );
}
